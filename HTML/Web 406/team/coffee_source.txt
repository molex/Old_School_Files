function getLowestPrice(aPrices) {

  // remove any ZERO entries
  aCleanPrices = new Array();
  for (i=1;i<aPrices.length;i++){
    if (aPrices[i] > 0){
      if (! ( (navigator.appVersion.indexOf("MSIE") != -1) && (navigator.userAgent.toLowerCase().indexOf("mac") != -1) ) ){
        aCleanPrices.push(aPrices[i]);
      }
    }
  }

  // if there's only one, use that
  if (aCleanPrices.length == 0)
    return ""
  else if (aCleanPrices.length == 1)
    return "$" + aCleanPrices[0];

  // get the minimum
  n=Math.min.apply(null,aCleanPrices);
           
  // make it a dollar figure
  if (n > 0) {
    var s = "" + Math.round(n * 100) / 100
    var i = s.indexOf('.')
    if (i < 0) return "From $" + s 
    var t = s.substring(0, i + 1) + s.substring(i + 1, i + 3)
    if (i + 2 == s.length) t += "0"
    return "From $" + t
  } else {
  return ""
  }

}
